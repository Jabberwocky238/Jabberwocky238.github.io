    package = package,
    class = class,
    function = function,

      Box::new(ret),
    )

  };

    None

  };

  quote! {

      env: JNIEnv<'local>,
      class: JClass<'local>,
      #(#args),*
    )  #ret {
      #function_before();
      #function(env, class, #(#args_),*  #comma_before_non_jni_args #(#non_jni_args),*)
    }

  }

}
